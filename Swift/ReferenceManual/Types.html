# 不透明类型

不透明类型可以理解为反向范型，范型是由调用者来决定具体类型，不透明类型是由实现来决定具体类型

不透明类型抛弃了协议的动态性，编译时就确定其具体类型

protocol View {
    associatedtype Body
    var body: Body { get }
}

struct AnyView<V>: View {
    typealias Body = V
    
    let view: V
    
    var body: Body {
        view
    }
}

func draw() -> some View {
    AnyView(view: "Hello, world!")
}

    func run() {
        self.start()

        let type: Self.Type = Self.self

  • # 内向外传递值

function f() {
    var a = 1
    return a
}

console.log(f())

    var a = 1
    callback(a)
}

f(a => {
    console.log(a)
})

    console.log(x, y, z)

# 10. async, await 和 Promise

function callback({ data, success, fail }) {
    setTimeout(() => {
        if (data.flag) {
            success('success')
        } else {
            fail('error')
        }
    }, 1000)
}

async function asyncPromise(data) {
    return await new Promise((resolve, reject) => {
        callback({
            data: {
                flag: data.flag
            },
            success: res => {
                resolve(res)
            },
            fail: err => {
                reject(err)
            }
        })
    })
}

(async () => {
    let res = await asyncPromise({
        flag: true
    })
    console.log(res)
})()
